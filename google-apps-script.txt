function doGet() {
  // Ganti dengan ID Spreadsheet Anda
  const spreadsheetId = "1pyTFglnajFZwy2oGxLxvkZ4kMcZ8EETrqNOyfDNXdL8";
  
  try {
    const ss = SpreadsheetApp.openById(spreadsheetId);
    const sheets = ss.getSheets();
    let allData = [];
    
    sheets.forEach(sheet => {
      const sheetName = sheet.getName().trim().toUpperCase();
      const data = sheet.getDataRange().getValues();
      if (data.length < 2) return;
      
      const headers = data[0].map(h => String(h).trim().toUpperCase());
      const rows = data.slice(1);
      
      const sheetData = rows.map(row => {
        const obj = {};
        headers.forEach((header, i) => {
          const val = row[i];
          
          // Map Indonesian headers to English keys expected by the app
          if (header === "TGL" || header === "DATE" || header === "TANGGAL") {
            obj["date"] = val;
          } else if (header === "URAIAN" || header === "DESCRIPTION" || header === "KETERANGAN") {
            obj["description"] = val;
          } else if (header === "JUMLAH" || header === "AMOUNT" || header === "NOMINAL") {
            obj["amount"] = val;
          } else if (header === "KAS" || header === "BANK" || header === "AKUN") {
            obj["bank"] = val;
          } else if (header === "POS" || header === "CATEGORY" || header === "KODE") {
            obj["category"] = val;
          } else if (header === "TRANSAKSI" || header === "TYPE" || header === "JENIS") {
            const typeStr = String(val).trim().toUpperCase();
            if (typeStr === "MASUK" || typeStr === "INCOME" || typeStr === "DEBIT") {
              obj["type"] = "Income";
            } else if (typeStr === "PENGELUARAN" || typeStr === "EXPENSE" || typeStr === "KREDIT") {
              obj["type"] = "Expense";
            } else {
              obj["type"] = val;
            }
          } else {
            obj[header] = val;
          }
        });

        // Fallback for bank if column is missing or empty
        if (!obj["bank"]) {
          obj["bank"] = sheet.getName();
        }

        return obj;
      });
      
      allData = allData.concat(sheetData);
    });

    return ContentService.createTextOutput(JSON.stringify(allData))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (e) {
    return ContentService.createTextOutput(JSON.stringify({ error: e.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
